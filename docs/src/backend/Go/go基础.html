package main

import "fmt"

func main() {

	/**
	len й•ҝеәҰ
	cap е®№йҮҸ
	*/

	var numbers = make([]int, 3, 5)
	fmt.Printf("len = %d, cap = %d, slice = %v\n\n", len(numbers), cap(numbers), numbers)
	// й•ҝеәҰ len = 3, е®№йҮҸ cap = 5, еҖј slice = [0 0 0]

	// еҗ‘numbersеҲҮзүҮиҝҪеҠ дёҖдёӘе…ғзҙ 1
	// len = 4, cap = 5, slice = [0 0 0 1]
	numbers = append(numbers, 1)
	fmt.Printf("len = %d, cap = %d, slice = %v\n\n", len(numbers), cap(numbers), numbers)

	// еҗ‘numbersеҲҮзүҮиҝҪеҠ дёҖдёӘе…ғзҙ 2
	// len = 5, cap = 5, slice = [0 0 0 1 2]
	numbers = append(numbers, 2)
	fmt.Printf("len = %d, cap = %d, slice = %v\n\n", len(numbers), cap(numbers), numbers)

	// еҗ‘numbersеҲҮзүҮиҝҪеҠ дёҖдёӘе…ғзҙ 3 и¶…иҝҮе®№йҮҸ
	// len = 6, cap = 10, slice = [0 0 0 1 2 3]
	numbers = append(numbers, 3)
	fmt.Printf("len = %d, cap = %d, slice = %v\n\n", len(numbers), cap(numbers), numbers)

	/**
	еҲҮзүҮжҲӘеҸ–
	*/
	s := []int{1, 2, 3} // len = 3 cap = 3
	s1 := s[0:2]        // [0, 2)
	fmt.Println(s1)     // [1 2]

	// жө…жӢ·иҙқ
	s1[0] = 100
	fmt.Println(s)  // [100 2 3]
	fmt.Println(s1) // [100 2]

	// ж·ұжӢ·иҙқ copyеҮҪж•°
	s2 := make([]int, 3) //[0, 0, 0]
	copy(s2, s)          // s --> s2
	fmt.Println(s2)      // [100 2 3]
}

package main

import "fmt"

type AnimalIF interface {
	Sleep()
	GetColor() string
	GetType() string
}

type Cat struct {
	color string // йўңиүІ
}

func (c *Cat) Sleep() {
	fmt.Println("Cat жӯЈеңЁзқЎи§ү")
}

func (c *Cat) GetColor() string {
	return c.color
}

func (c *Cat) GetType() string {
	return "Cat"
}

type Dog struct {
	color string
}

func (d *Dog) Sleep() {
	fmt.Println("Dog жӯЈеңЁзқЎи§ү")
}

func (d *Dog) GetColor() string {
	return d.color
}

func (d *Dog) GetType() string {
	return "Dog"
}

// дј иҝӣжқҘзҡ„жҳҜ *Cat е°ұиө° Cat зҡ„ж–№жі•пјҢдј иҝӣжқҘзҡ„жҳҜ *Dog е°ұиө° Dog зҡ„ж–№жі• вҖ”вҖ” иҝҷе°ұжҳҜеӨҡжҖҒ
func showAnimal(a AnimalIF) {
	fmt.Printf("зұ»еһӢ: %s, йўңиүІ: %s\n", a.GetType(), a.GetColor())
	a.Sleep()
}

func main() {
	cat := &Cat{color: "red"}
	dog := &Dog{color: "black"}

	// еҗҢдёҖдёӘ showAnimal еҮҪж•°пјҢдј е…ҘдёҚеҗҢе®һзҺ°пјҢиЎҢдёәдёҚеҗҢ
	showAnimal(cat) // зұ»еһӢ: Cat, йўңиүІ: red / Cat жӯЈеңЁзқЎи§ү
	showAnimal(dog) // зұ»еһӢ: Dog, йўңиүІ: black / Dog жӯЈеңЁзқЎи§ү
}

package main

import (
	"fmt"
	"reflect"

func reflectNum(arg interface{}) {
	fmt.Println(reflect.TypeOf(arg))  // type
	fmt.Println(reflect.ValueOf(arg)) // value
}

type User struct {
	Id   string
	Name string
	Age  int
}

func (u User) Call() {
	fmt.Println("user is called ...")
	fmt.Println(u)
}

func main() {
	//var num float64 = 1.23456
	//reflectNum(num)

	user := User{"1", "Jack", 34}
	DoFileAndMethod(user)
}

func DoFileAndMethod(input interface{}) {
	// иҺ·еҸ–inputзҡ„Type
	inputType := reflect.TypeOf(input)
	fmt.Println(inputType.Name()) // User
	fmt.Println(inputType)        // main.User

	// иҺ·еҸ–inputзҡ„Value
	inputValue := reflect.ValueOf(input)
	fmt.Println(inputValue) // {1 Jack 34}

	// йҖҡиҝҮtype иҺ·еҸ–йҮҢйқўзҡ„еӯ—ж®ө
	// 1. иҺ·еҸ–interfaceзҡ„reflect.type йҖҡиҝҮtypeеҫ—еҲ°NumField иҝӣиЎҢйҒҚеҺҶ
	// 2.еҫ—еҲ°жҜҸдёӘfield ж•°жҚ®зұ»еһӢ
	// 3. fieldзҡ„Interface()ж–№жі• еҫ—еҲ°еҜ№еә”зҡ„value
	for i := 0; i < inputType.NumField(); i++ {
		filed := inputType.Field(i)
		value := inputValue.Field(i).Interface()
		fmt.Println("----------------------")
		fmt.Println(filed.Name, filed.Type, value)
	}

	// йҖҡиҝҮtype иҺ·еҸ–йҮҢйқўзҡ„ж–№жі• и°ғз”Ё
	for i := 0; i < inputType.NumMethod(); i++ {
		method := inputType.Method(i)
		fmt.Println("********************")
		fmt.Println(method.Name, method.Type)
	}
}

	// Elem() е°ұжҳҜи§Јеј•з”ЁпјҢеҺ»жҺүжҢҮй’Ҳзҡ„ *
	t := reflect.TypeOf(str).Elem()

		fmt.Println("doc:", tagdoc)
	}
}

	var re resume
	findTag(&re)
}

go
package main

import "fmt"

// T жҳҜзұ»еһӢеҸӮж•°пјҢany иЎЁзӨәгҖҢд»»ж„Ҹзұ»еһӢгҖҚ
func Reverse[T any](s []T) []T {
	for i, j := 0, len(s)-1; i < j; i, j = i+1, j-1 {
		s[i], s[j] = s[j], s[i]
	}
	return s
}

func main() {
	// и°ғз”Ёж—¶еҸҜд»ҘзңҒз•Ҙзұ»еһӢеҸӮж•°пјҢGo дјҡиҮӘеҠЁжҺЁж–ӯ
	fmt.Println(Reverse([]int{1, 2, 3}))       // [3 2 1]
	fmt.Println(Reverse([]string{"a", "b", "c"})) // [c b a]
}

	data []T

func (s *Stack[T]) Push(v T) {
	s.data = append(s.data, v)
}

	var zero T
	if len(s.data) == 0 {

	}
	v := s.data[len(s.data)-1]
	s.data = s.data[:len(s.data)-1]
	return v, true
}
