### 2. **部署脚本改进** ⭐⭐⭐⭐⭐

**当前问题**：

```javascript
// index.js 存在的问题：
1. 使用已废弃的 fs.rmdirSync()
2. 错误处理不足
3. spinner 启动后立即开始执行，没有实际意义
4. 缺少参数校验
5. 硬编码分支名称 'main'
```

**优化方案**：

```javascript
import { execSync } from "child_process";
import ora from "ora";
import fs from "node:fs";

// 清理构建目录
const clearFolder = () => {
  if (fs.existsSync("./docs")) {
    fs.rmSync("./docs", { recursive: true, force: true }); // 使用新API
  }
};

// 执行 git 命令
const gitCommander = (command) => {
  try {
    execSync(`git ${command}`, { stdio: "inherit" });
  } catch (error) {
    console.error(`Git 命令执行失败: ${command}`);
    throw error;
  }
};

// 主构建流程
function build() {
  const commitMessage = process.argv[2];

  // 参数校验
  if (!commitMessage) {
    console.error("❌ 错误: 缺少 commit 信息");
    console.log('使用方法: npm run deploy "your commit message"');
    process.exit(1);
  }

  try {
    clearFolder();

    // 构建
    let spinner = ora("正在构建项目...").start();
    execSync("npm run build", { stdio: "inherit" });
    spinner.succeed("✅ 构建完成");

    // Git 操作
    spinner = ora("正在提交代码...").start();
    gitCommander("add .");
    gitCommander(`commit -m "${commitMessage}"`);
    spinner.succeed("✅ 代码提交完成");

    // 推送
    spinner = ora("正在推送到远程仓库...").start();
    const branch = execSync("git branch --show-current").toString().trim();
    gitCommander(`push origin ${branch}`);
    spinner.succeed(`✅ 已推送到 ${branch} 分支`);

    console.log("\n🎉 部署流程执行完成！");
  } catch (error) {
    console.error("\n❌ 部署失败:", error.message);
    process.exit(1);
  }
}

build();
```

### 3. **项目结构优化** ⭐⭐⭐

**当前问题**：

- 部署脚本 `index.js` 散落在根目录
- `stats.html` 等分析文件混在根目录

> ⚠️ **注意：本项目的 `docs/` 是构建输出目录，不是源码目录。**
> 源码内容位于根目录的 `src/`，VitePress 通过 `outDir: "docs"` 把产物输出到 `docs/` 供 GitHub Pages 发布。这与 VitePress 官方「源码放 `docs/`、产物输出到 `docs/.vitepress/dist`」的默认布局相反，**重构时切勿把源码移进 `docs/`**，否则会与 `outDir` 配置冲突并污染发布目录。

**可选的轻量调整**（不影响部署）：

```
项目根目录/
├── .vitepress/          # VitePress 配置
│   ├── config.mts
│   └── cache/
├── src/                 # 📝 文档源码（实际内容在这里）
│   ├── frontend/
│   ├── backend/
│   ├── web/
│   └── other/
├── public/              # 静态资源
├── docs/                # ⚙️ 构建输出（GitHub Pages 发布目录，需提交）
├── scripts/             # 🆕 可选：把 index.js 移到这里改名 deploy.js
│   └── deploy.js
├── .gitignore
├── package.json
├── index.md             # 首页
├── 简介.md
└── 主要内容概述.md
```

> 若把 `index.js` 移到 `scripts/deploy.js`，记得同步修改 `package.json` 的 `deploy` 脚本路径。

### 9. **CI/CD 优化** ⭐⭐⭐

> ⚠️ **不要照抄常见的 `peaceiris/actions-gh-pages` 示例！**
> 那种方案会把产物推到独立的 `gh-pages` 分支，而你现在是「`main` 分支 + `/docs` 文件夹」发布。两套机制混用会互相打架。**要上 CI/CD，必须先二选一**，不能直接套用。

**两条可行路线（任选其一）**：

**路线 A：保持现有「main 分支 /docs 文件夹」模式，仅自动化构建提交**

```yaml
name: Build and Deploy Docs

on:
  push:
    branches: [main]
    paths-ignore:
      - "docs/**" # 避免提交产物后再次触发，造成死循环

permissions:
  contents: write

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - run: npm run build
      - name: Commit built docs
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add docs
          git diff --staged --quiet || git commit -m "chore: rebuild docs [skip ci]"
          git push
```

> 采用路线 A 后，本地就不必再跑 `npm run deploy`，但要注意循环触发问题（已用 `paths-ignore` 规避）。

**路线 B：改用 GitHub 官方 Pages Actions（推荐，产物不再入库）**

切换为「GitHub Actions」发布源（仓库 Settings → Pages → Source 改为 GitHub Actions），此时 `docs/` 可以不再提交：

```yaml
name: Deploy to GitHub Pages

on:
  push:
    branches: [main]

permissions:
  contents: read
  pages: write
  id-token: write

jobs:
  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
      - run: npm ci
      - run: npm run build
      - uses: actions/configure-pages@v4
      - uses: actions/upload-pages-artifact@v3
        with:
          path: docs
      - id: deployment
        uses: actions/deploy-pages@v4
```

> ⚠️ 仅在采用路线 B 后，才可以把 `docs/` 加入 `.gitignore`；否则保持提交 `docs/`。
